function getHeader(msg, name) { return (msg.payload?.headers||[]).find(h=>h.name===name)?.value || ''; }
function formatDate(d) {
  if (!d) return '';
  const dt = new Date(d), now = new Date();
  return dt.toDateString()===now.toDateString() ? '今天' : (dt.getMonth()+1)+'/'+dt.getDate();
}
function classifyEmail(subject, from) {
  if (keywords.some(k=>subject.includes(k)) || senderFilters.some(s=>from.toLowerCase().includes(s.toLowerCase()))) return 'course';
  if (/發票|對帳單|訂單|付款|申購/.test(subject)) return 'finance';
  if (/render|github|railway|failed|exceeded/i.test(from)) return 'urgent';
  if (/pinterest|gaiconf|readingoutpost|notion.*mail/i.test(from) || /廣告|電子報|推薦|特訓|活動|小聚/i.test(subject)) return 'ad';
  return 'info';
}
function typeLabel(t) { return {urgent:'緊急',course:'課程偵測',finance:'財務',ad:'廣告',info:'一般'}[t]||t; }
function esc(s) { return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }

function renderInbox(messages) {
  const groups = {urgent:[],course:[],finance:[],info:[],ad:[]};
  messages.forEach(msg => {
    const subject = getHeader(msg,'Subject')||'（無主旨）';
    const from = getHeader(msg,'From')||'';
    const date = formatDate(getHeader(msg,'Date'));
    const type = classifyEmail(subject, from);
    groups[type].push({msg,subject,from,date,type});
  });
  const labels = {urgent:['⚠️','需要關注'],course:['📢','課程偵測'],finance:['💰','財務帳務'],info:['📨','一般信件'],ad:['🗑','廣告 / 電子報']};
  let html = '';
  Object.entries(labels).forEach(([type,[icon,label]])=>{
    if (!groups[type].length) return;
    html += `<div class="sec-header"><div class="sec-title">${icon} ${label}</div></div>`;
    groups[type].forEach(({msg,subject,from,date,type})=>{
      const sender = from.replace(/<[^>]+>/,'').replace(/"/g,'').trim() || from;
      html += `<div class="mail-card ${type}">
        <div class="mail-top"><div class="mail-sender">${esc(sender.substring(0,22))}</div><div class="mail-date">${date}</div></div>
        <div class="mail-subject">${esc(subject.substring(0,52))}</div>
        <div class="mail-snippet">${esc((msg.snippet||'').substring(0,60))}</div>
        <div class="mail-tags"><span class="tag ${type}">${typeLabel(type)}</span></div>
        <div class="mail-actions">
          <button class="btn-sm btn-ghost" onclick="markRead('${msg.id}',this)">標記已讀</button>
          <button class="btn-sm btn-danger" onclick="trashMail('${msg.id}',this)">🗑 刪除</button>
        </div>
      </div>`;
    });
  });
  if (!html) html = '<div class="empty-state"><div class="empty-icon">🎉</div>沒有未讀信件！</div>';
  document.getElementById('inbox-content').innerHTML = html;
}

}
async function trashMail(id, btn) {

}

async function doForward(msgId, subject) {
  showToast('發送中，請稍候...');
  const usePrefix = document.getElementById('toggle-prefix').classList.contains('on');
  const finalSubject = (usePrefix ? '【轉發】' : '') + subject;

  let originalBody = '';

    const fullMsg = await gmailFetch(`https://www.googleapis.com/gmail/v1/users/me/messages/${msgId}?format=full`);
    const findBody = (part) => {
      if (!part) return '';
      if (part.mimeType === 'text/html' && part.body?.data) return atob(part.body.data.replace(/-/g,'+').replace(/_/g,'/'));
      if (part.mimeType === 'text/plain' && part.body?.data) return '<pre>' + atob(part.body.data.replace(/-/g,'+').replace(/_/g,'/')) + '</pre>';
      if (part.parts) { for (const p of part.parts) { const r = findBody(p); if (r) return r; } }
      return '';
    };
    originalBody = findBody(fullMsg.payload) || `<p>${fullMsg.snippet||''}</p>`;
  } catch(e) { originalBody = '<p>（無法載入原始信件內容）</p>'; }

  const emailHtml = `
<div style="font-family:'Noto Sans TC',Arial,sans-serif;font-size:14px;color:#333;line-height:1.8;">
  <p>各位會員好：</p>
  <p>如標題附件，請參考，謝謝。</p>
  <hr style="border:none;border-top:1px solid #ddd;margin:16px 0;">
  ${originalBody}

    社團法人澎湖縣職能治療師公會<br>
    880澎湖縣馬公市光榮里縣府新村23號1樓<br>

    總幹事 林蘭蕙<br>
    電話：0932463513<br>
    E-mail：<a href="mailto:pehotu.office@gmail.com">pehotu.office@gmail.com</a>
  </p>
</div>`;

  let ok = 0;
  for (const m of members) {
    try {
      const raw = makeEmail(m.email, finalSubject, emailHtml);
      await gmailFetch('https://www.googleapis.com/gmail/v1/users/me/messages/send',{method:'POST',body:JSON.stringify({raw})});

}

function makeEmail(to, subject, html) {
  const msg = ['To: '+to,'From: '+userEmail,'Subject: =?UTF-8?B?'+btoa(unescape(encodeURIComponent(subject)))+'?=','MIME-Version: 1.0','Content-Type: text/html; charset=UTF-8','',html].join('\r\n');
  return btoa(unescape(encodeURIComponent(msg))).replace(/\+/g,'-').replace(/\//g,'_').replace(/=+$/,'');
}

function loadCSV(e) {
  const file = e.target.files[0]; if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    const lines = ev.target.result.split('\n').filter(l=>l.trim());
    members = [];
    lines.forEach(line=>{
      const p = line.split(',');
      if (p.length>=2) { const name=p[0].trim().replace(/"/g,''), email=p[1].trim().replace(/"/g,''); if (email.includes('@')) members.push({name,email}); }
    });
    document.getElementById('member-count').textContent = members.length;
    renderMembers(); showToast(`✅ 匯入 ${members.length} 位會員`);
  };
  reader.readAsText(file,'UTF-8');
}

}

    const type = classifyEmail(subject, from);

        <div class="mail-date">${date}</div>
      </div>

      <div class="mail-snippet">${esc((msg.snippet||'').substring(0,60))}</div>

}

let searchTimer;
async function searchEmails(query) {
  clearTimeout(searchTimer);
  if (!query.trim()) {
    if (allEmailsCache.length) renderAllEmails(allEmailsCache);
    return;
  }
  searchTimer = setTimeout(async () => {
    document.getElementById('all-loading').style.display = 'flex';
    document.getElementById('all-content').innerHTML = '';
    try {
      const data = await gmailFetch(`https://www.googleapis.com/gmail/v1/users/me/messages?labelIds=INBOX&q=${encodeURIComponent(query)}&maxResults=20`);
      const messages = data.messages || [];
      if (!messages.length) {
        document.getElementById('all-content').innerHTML = '<div class="empty-state"><div class="empty-icon">🔍</div>找不到相關信件</div>';
        document.getElementById('all-loading').style.display = 'none';
        return;
      }
      const details = await Promise.all(messages.map(m =>
        gmailFetch(`https://www.googleapis.com/gmail/v1/users/me/messages/${m.id}?format=metadata&metadataHeaders=Subject&metadataHeaders=From&metadataHeaders=Date`)
      ));
      renderAllEmails(details);
    } catch(e) { showToast('❌ 搜尋失敗'); }
    document.getElementById('all-loading').style.display = 'none';
  }, 500);
}
